from dataclasses import dataclass
from queue import Queue

@dataclass
class Node:
    data: int
    left: object = None
    right: object = None
    
    
def bfs(root: Node):
    if root is None:
        return
    
    q = Queue()
    q.put(root)
    
    while not q.empty():
        curr = q.get()
        if curr.left is not None:
            q.put(curr.left)
        
        if curr.right is not None:
            q.put(curr.right)
            
        print(curr.data)
        

t = Node(1, Node(2, Node(3)), Node(4))
bfs(t)

class AdjUndirectedGraph:
    def __init__(self):
        self.adj_list = dict() # hashmap
    
    def add(self, a,b):
        
        if a not in self.adj_list:
            self.adj_list[a] = []
        
        self.adj_list[a].append(b)
        
        if b not in self.adj_list:
            self.adj_list[b] = []
        self.adj_list[b].append(a)
        
    def print(self):
        for k,v in self.adj_list.items():
            print(k, v)
            
    def breadth_first_traversal(self):
        if len(self.adj_list) == 0:
            return
        
        visited = set()
        
        q = Queue()
        v = next(iter(self.adj_list.keys())) # pick any one vertex
        q.put(v)
        visited.add(v)

        while not q.empty():
            curr = q.get() # graph specific
            
            # add all adjacent nodes of curr to the q
            for v in self.adj_list[curr]:
                if v not in visited: # graph specific
                    visited.add(v)
                    q.put(v)
            
            print(curr)
