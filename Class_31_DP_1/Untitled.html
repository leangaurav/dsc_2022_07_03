func fibonacciSequence (n: Int) -> [Int]  {
    var fibonacciArray = [Int]()

    for n in 0 ... n {

        if n == 0 {
            fibonacciArray.append(0)
        }
        else if n == 1 {
            fibonacciArray.append(1)
        }
        else {
            fibonacciArray.append (fibonacciArray[n-1] + fibonacciArray[n-2] )
        }
    }
    return fibonacciArray
}

# TC: O(N)
# SC: O(N)
class Solution:
    def climbStairs(self, n: int) -> int:
        arr = {0:1,1:2}
        for i in range(1,n):
            arr[i+1] = arr[i]+arr[i-1]
        return arr[n-1]

// TC: O(N)
// SC: O(N)

var climbStairs = function(n,map = {}) {
    if(n === 1) {
        return 1;
    } else if(n === 2) {
        return 2;
    }

    if(!map[n]) {
        map[n] = climbStairs(n-1, map) + climbStairs(n-2, map);
    }  
    return map[n];
};
